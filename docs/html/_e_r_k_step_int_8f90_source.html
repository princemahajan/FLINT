120 subroutine stepint(pDiffEqSys, X0, Y0, F0, h, IsFSALMethod, sint, k, a, b, c, Y1, &
121 Yint12, FCalls, Err, IsScalarTol, RTol, ATol, e, params)
122 class(DiffEqSys), intent(inout) :: pDiffEqSys
123 real(WP), intent(in) :: X0
124 real(WP), dimension(pDiffEqSys%n), intent(in) :: Y0
125 real(WP), dimension(pDiffEqSys%n), intent(inout) :: F0
126 real(WP), intent(in) :: h
127 logical, intent(in) :: IsFSALMethod
128 integer, intent(in) :: sint
129 real(WP), dimension(size(Y0),1:sint), intent(inout) :: k
130 real(WP), dimension(:), contiguous, intent(in) :: a
131 real(WP), dimension(1:sint), intent(in) :: b
132 real(WP), dimension(2:sint), intent(in) :: c
133 real(WP), dimension(size(Y0)), intent(out) :: Y1, Yint12
134 integer, intent(out) :: FCalls
135 real(WP), intent(out) :: Err
136 logical, intent(in), optional :: IsScalarTol
137 real(WP), dimension(size(Y0)), intent(in), optional :: RTol, ATol
138 real(WP), dimension(1:sint), intent(in), optional :: e
139 real(WP), dimension(:), intent(in), optional :: params
140
141 real(WP), dimension(size(Y0)) :: Yint
142 integer :: i, n
143 logical :: EstimateErr
144
145 estimateerr = .false.
146 if (present(isscalartol) .AND. present(rtol) &
147 .AND. present(atol) .AND. present(e)) estimateerr = .true.
148
149 n = pdiffeqsys%n
150
151 ! first stage (we expect F0 already contains the initial derivative)
152 k(:,1) = f0
153
154 ! compute rest of the stages
155 ! This compact and beautiful code is slower than ugly hardcoded one!
156 do i = 2,sint
157 block
158 integer :: astart, j
159 real(WP), dimension(n) :: aijkj
160 ! starting index of a_ij, where i=i, j=1:(i-1)
161 ! It is a series sum: 1 + 2 + 3 + ...
162 astart = int((i-1)/2.0*(i-2))
163 ! In my testing, I am seeing matmul performing better than
164 ! MKL's gemv. However, do loop is faster than both but
165 ! but still slower than hard-coded computations
166 ! call gemv(k(:,1:(i-1)), me%a((astart+1):(i-1)), aijkj)
167 ! aijkj = matmul(k(:,1:(i-1)), a((astart+1):(astart+i-1)))
168 aijkj = 0.0_wp
169 do j = 1,(i-1)
170 aijkj = aijkj + k(:,j)*a(astart + j)
171 end do
172
173 yint = y0 + h*aijkj
174 k(:,i) = pdiffeqsys%F(x0 + h*c(i), yint, params)
175
176 ! Yint for the last non-FSAL stage is needed for stiffness detection
177 ! This is from Hairer's DOP853, so not sure its correct for other solvers
178 ! TBD: I dont like this, need a way to avoid this condition in this loop
179 if (i == (sint-1)) yint12 = yint
180 end block
181 end do
182
183 ! propagate the solution to the next step for error computation
184 ! For FSAL methods, Yint already has the new solution
185
186 ! For non-FSAL methods, compute the solution at the next step
187 if (isfsalmethod .EQV. .false.) then
188 ! for stiffness detection, we need last non-FSAL stage solution
189 ! TBD: this needs correction for stifness detection in case of
190 ! non-FSAL methods as stiffness test needs Yint12 and F(Yint12)
191 ! and we replaced Yint12 with Yint but still have old F(Yint12)
192 yint12 = yint
193 yint = y0 + h*matmul(k(:,1:sint), b(1:sint))
194 end if
195
196 if (estimateerr) then
197 ! Estimate the error. We assume that the error coeffcients are precomputed,
198 ! i.e., e_i = b_i - bhat_i
199 block
200 real(WP), dimension(n) :: temp, Sc
201 integer :: j
202 ! The scale factor for error computations
203 if (isscalartol .EQV. .false.) then
204 sc = atol + rtol*max(abs(y0), abs(yint))
205 else
206 sc = atol(1) + rtol(1)*max(abs(y0), abs(yint))
207 end if
208 temp = 0.0_wp
209 do concurrent(j = 1:sint)
210 temp = temp + k(:,j)*e(j)
211 end do
212 err = sqrt(sum((h*temp/sc)**2)/n)
213 end block
214 else
215 err = 0.0_wp
216 end if
217
218 ! If this step is accepted then return the new solution
219 ! else just return the initial condition
220 if (err <= 1.0_wp) then
221 y1 = yint
222 if (isfsalmethod .EQV. .false.) then
223 ! Evaluate F0 for the next step
224 f0 = pdiffeqsys%F(x0 + h, y1, params)
225 ! Number of function calls made so far
226 fcalls = sint
227 else
228 ! In FSAL methods, the integration last stage is F0 for the next step
229 f0 = k(:,sint)
230 ! Function calls made if the step is accepted
231 fcalls = sint - 1
232 end if
233 else
234 y1 = y0
235 ! Function calls made if the step is rejected
236 if (isfsalmethod .EQV. .false.) then
237 fcalls = sint - 1
238 else
239 fcalls = sint - 2
240 end if
241 end if
242 end subroutine stepint

245 subroutine intpcoeff(pDiffEqSys, X0, Y0, h, InterpStates, sint, s, a, c, d, &
246 dinz, pstar, k, IpCoeff, FCalls, params)
247 class(DiffEqSys), intent(inout) :: pDiffEqSys
248 real(WP), intent(in) :: X0
249 real(WP), dimension(pDiffEqSys%n), intent(in) :: Y0
250 real(WP), intent(in) :: h
251 !real(WP), dimension(size(Y0)), intent(in) :: Y1
252 integer, dimension(:), intent(in) :: InterpStates
253 integer, intent(in) :: sint, s
254 real(WP), dimension(:), contiguous, intent(in) :: a
255 real(WP), dimension(sint+1:s), intent(in) :: c
256 real(WP), dimension(:,:), contiguous, intent(in) :: d
257 integer, dimension(:), contiguous, intent(in) :: dinz
258 integer, intent(in) :: pstar
259 real(WP), dimension(size(Y0),1:s), intent(inout) :: k
260 real(WP), dimension(size(InterpStates),0:pstar), intent(out) :: IpCoeff
261 integer, intent(out) :: Fcalls
262 real(WP), dimension(:), intent(in), optional :: params
263
264 integer :: i, j
265
266 ! Compute extra stages needed for interpolation
267
268 ! compute rest of the stages (the following compact code is slower!)
269 do i = (sint+1),s
270 block
271 integer :: astart, j
272 real(WP), dimension(size(Y0)) :: aijkj
273 ! starting index of a_ij, where i=i, j=1:(i-1)
274 ! It is a series sum: 1 + 2 + 3 + ...
275 astart = int((i-1)/2.0*(i-2))
276 ! In my testing, I am seeing matmul performing better than
277 ! MKL's gemv. However, do concurrent is faster than both but
278 ! but still slower than hard-coded computations
279 ! call gemv(me%k(:,1:(i-1)), me%a((astart+1):(i-1)), aijkj)
280 ! aijkj = matmul(me%k(:,1:(i-1)), me%a((astart+1):(astart+i-1)))
281 aijkj = 0.0_wp
282 do concurrent(j = 1:(i-1))
283 aijkj = aijkj + k(:,j)*a(astart + j)
284 end do
285
286 aijkj = y0 + h*aijkj
287 k(:,i) = pdiffeqsys%F(x0 + h*c(i), aijkj, params)
288 end block
289 end do
290
291 fcalls = s - sint
292
298
299 ipcoeff(:,0) = y0(interpstates)
300 do concurrent(j = 1:pstar)
301 block
302 integer :: istage
303 real(WP), dimension(size(InterpStates)) :: cont
304
305 cont = 0.0_wp
306 sloop: do i = 1,s
307 ! get the current stage that gets multiplied by non-zero dij
308 istage = dinz(i)
309 ! -1 means we are at the end of the list
310 if (istage == -1) exit sloop
311
312 cont = cont + d(i,j)*k(interpstates,istage)
313 end do sloop
314 ipcoeff(:,j) = h*cont
315
316 end block
317 end do
318 end subroutine intpcoeff

322 subroutine dop853_stepint(pDiffEqSys, X0, Y0, F0, h, k, Y1, Yint12, FCalls, Err, &
323 IsScalarTol, RTol, ATol, params)
324 class(DiffEqSys), intent(inout) :: pDiffEqSys
325 real(WP), intent(in) :: X0
326 real(WP), dimension(pDiffEqSys%n), intent(in) :: Y0
327 real(WP), dimension(pDiffEqSys%n), intent(inout) :: F0
328 real(WP), intent(in) :: h
329 real(WP), dimension(size(Y0),1:DOP853_sint), intent(inout) :: k
330 real(WP), dimension(size(Y0)), intent(out) :: Y1, Yint12
331 integer, intent(out) :: FCalls
332 real(WP), intent(out) :: Err
333 logical, intent(in), optional :: IsScalarTol
334 real(WP), dimension(size(Y0)), intent(in), optional :: RTol, ATol
335 real(WP), dimension(:), intent(in), optional :: params
336
337 real(WP), dimension(size(Y0)) :: Yint, aijkj, Sc
338 logical EstimateErr
339 real(WP) :: DenomErr, Err3
340
341 estimateerr = .false.
342 if (present(isscalartol) .AND. present(rtol) &
343 .AND. present(atol)) estimateerr = .true.
344
345
346 ! first stage (we expect F0 already contains the initial derivative)
347 k(:,1) = f0
348
349 ! 2nd stage
350 aijkj = k(:,1)*dop853_a(1)
351 yint = y0 + h*aijkj
352 k(:,2) = pdiffeqsys%F(x0 + h*dop853_c(2), yint, params)
353
354 ! 3rd stage
355 aijkj = k(:,1)*dop853_a(2) + k(:,2)*dop853_a(3)
356 yint = y0 + h*aijkj
357 k(:,3) = pdiffeqsys%F(x0 + h*dop853_c(3), yint, params)
358
359 ! 4th stage
360 aijkj = k(:,1)*dop853_a(4) + k(:,3)*dop853_a(6)
361 yint = y0 + h*aijkj
362 k(:,4) = pdiffeqsys%F(x0 + h*dop853_c(4), yint, params)
363
364 ! 5th stage
365 aijkj = k(:,1)*dop853_a(7) + k(:,3)*dop853_a(9) + k(:,4)*dop853_a(10)
366 yint = y0 + h*aijkj
367 k(:,5) = pdiffeqsys%F(x0 + h*dop853_c(5), yint, params)
368
369 ! 6th stage
370 aijkj = k(:,1)*dop853_a(11) + k(:,4)*dop853_a(14) &
371 + k(:,5)*dop853_a(15)
372 yint = y0 + h*aijkj
373 k(:,6) = pdiffeqsys%F(x0 + h*dop853_c(6), yint, params)
374
375 ! 7th stage
376 aijkj = k(:,1)*dop853_a(16) + k(:,4)*dop853_a(19) &
377 + k(:,5)*dop853_a(20) + k(:,6)*dop853_a(21)
378 yint = y0 + h*aijkj
379 k(:,7) = pdiffeqsys%F(x0 + h*dop853_c(7), yint, params)
380
381 ! 8th stage
382 aijkj = k(:,1)*dop853_a(22) + k(:,4)*dop853_a(25) &
383 + k(:,5)*dop853_a(26) + k(:,6)*dop853_a(27) + k(:,7)*dop853_a(28)
384 yint = y0 + h*aijkj
385 k(:,8) = pdiffeqsys%F(x0 + h*dop853_c(8), yint, params)
386
387 ! 9th stage
388 aijkj = k(:,1)*dop853_a(29) + k(:,4)*dop853_a(32) &
389 + k(:,5)*dop853_a(33) + k(:,6)*dop853_a(34) + k(:,7)*dop853_a(35) + k(:,8)*dop853_a(36)
390 yint = y0 + h*aijkj
391 k(:,9) = pdiffeqsys%F(x0 + h*dop853_c(9), yint, params)
392
393 ! 10th stage
394 aijkj = k(:,1)*dop853_a(37) + k(:,4)*dop853_a(40) &
395 + k(:,5)*dop853_a(41) + k(:,6)*dop853_a(42) + k(:,7)*dop853_a(43) + k(:,8)*dop853_a(44) &
396 + k(:,9)*dop853_a(45)
397 yint = y0 + h*aijkj
398 k(:,10) = pdiffeqsys%F(x0 + h*dop853_c(10), yint, params)
399
400 ! 11th stage
401 aijkj = k(:,1)*dop853_a(46) + k(:,4)*dop853_a(49) &
402 + k(:,5)*dop853_a(50) + k(:,6)*dop853_a(51) + k(:,7)*dop853_a(52) + k(:,8)*dop853_a(53) &
403 + k(:,9)*dop853_a(54) + k(:,10)*dop853_a(55)
404 yint = y0 + h*aijkj
405 k(:,11) = pdiffeqsys%F(x0 + h*dop853_c(11), yint, params)
406
407 ! 12th stage
408 aijkj = k(:,1)*dop853_a(56) + k(:,4)*dop853_a(59) &
409 + k(:,5)*dop853_a(60) + k(:,6)*dop853_a(61) + k(:,7)*dop853_a(62) + k(:,8)*dop853_a(63) &
410 + k(:,9)*dop853_a(64) + k(:,10)*dop853_a(65) + k(:,11)*dop853_a(66)
411 yint12 = y0 + h*aijkj
412 k(:,12) = pdiffeqsys%F(x0 + h*dop853_c(12), yint12, params)
413
414 ! 13th stage
415 aijkj = k(:,1)*dop853_a(67) &
416 + k(:,6)*dop853_a(72) + k(:,7)*dop853_a(73) + k(:,8)*dop853_a(74) &
417 + k(:,9)*dop853_a(75) + k(:,10)*dop853_a(76) + k(:,11)*dop853_a(77) + k(:,12)*dop853_a(78)
418 yint = y0 + h*aijkj
419
420 ! propagate the solution to the next step for error computation
421 ! For FSAL, Yint is the new solution
422
423 if (estimateerr) then
424 ! The scale factor for error computations
425 if (isscalartol .EQV. .false.) then
426 sc = atol + rtol*max(abs(y0), abs(yint))
427 else
428 sc = atol(1) + rtol(1)*max(abs(y0), abs(yint))
429 end if
430
431 ! Estimate the error. We assume that the error coeffcients are precomputed,
432 ! i.e., e_i = b_i - bhat_i
433
434 err = sum(((k(:,1)*dop853_e(1) + k(:,6)*dop853_e(6) + k(:,7)*dop853_e(7) + k(:,8)*dop853_e(8) &
435 + k(:,9)*dop853_e(9) + k(:,10)*dop853_e(10) + k(:,11)*dop853_e(11) + k(:,12)*dop853_e(12))/sc)**2)
436
437 ! For DOP853, we need to apply a 3rd-order correction
438 err3 = sum(((aijkj-dop853_bhh(1)*k(:,1)-dop853_bhh(2)*k(:,9) &
439 - dop853_bhh(3)*k(:,12))/sc)**2)
440
441 denomerr = err + 0.01_wp*err3
442 if (denomerr == 0.0_wp) denomerr = 1.0_wp
443 err = abs(h)*err*sqrt(1.0/(pdiffeqsys%n*denomerr))
444 else
445 err = 0.0_wp
446 end if
447
448
449 ! If this step is accepted then return the new solution
450 ! else just return the initial condition
451 if (err <= 1.0_wp) then
452 y1 = yint
453 ! In FSAL methods, the intgeration last stage is F0 for the next step
454 k(:,13) = pdiffeqsys%F(x0 + h, yint, params)
455 f0 = k(:,13)
456 ! Function calls made if the step is accepted
457 fcalls = dop853_sint - 1
458 else
459 y1 = y0
460 ! Function calls made if the step is rejected
461 fcalls = dop853_sint - 2
462 end if
463 end subroutine dop853_stepint

466 subroutine dop853_intpcoeff(pDiffEqSys, X0, Y0, h, Y1, InterpStates, k, IntpCoeff, Fcalls, params)
467 class(DiffEqSys), intent(inout) :: pDiffEqSys
468 real(WP), intent(in) :: X0
469 real(WP), dimension(pDiffEqSys%n), intent(in) :: Y0
470 real(WP), intent(in) :: h
471 real(WP), dimension(size(Y0)), intent(in) :: Y1
472 integer, dimension(:), intent(in) :: InterpStates
473 real(WP), dimension(:,:), intent(inout) :: k
474 real(WP), dimension(size(InterpStates),0:7), intent(out) :: IntpCoeff
475 integer, intent(out) :: Fcalls
476 real(WP), dimension(:), intent(in), optional :: params
477
478 real(WP), dimension(pDiffEqSys%n) :: Yint, aijkj
479 real(WP), dimension(size(InterpStates)) :: cont0, cont1, cont2, cont3, cont4, cont5, cont6, cont7
480
481 ! Compute extra stages needed for interpolation
482
483 ! 14th stage
484 aijkj = k(:,1)*dop853_a(79) + k(:,7)*dop853_a(85) + k(:,8)*dop853_a(86) + k(:,9)*dop853_a(87) &
485 + k(:,10)*dop853_a(88) + k(:,11)*dop853_a(89) + k(:,12)*dop853_a(90) + k(:,13)*dop853_a(91)
486 yint = y0 + h*aijkj
487 k(:,14) = pdiffeqsys%F(x0 + h*dop853_c(14), yint, params)
488
489 ! 15th stage
490 aijkj = k(:,1)*dop853_a(92) + k(:,6)*dop853_a(97) + k(:,7)*dop853_a(98) + k(:,8)*dop853_a(99) &
491 + k(:,11)*dop853_a(102) + k(:,12)*dop853_a(103) + k(:,13)*dop853_a(104) + k(:,14)*dop853_a(105)
492 yint = y0 + h*aijkj
493 k(:,15) = pdiffeqsys%F(x0 + h*dop853_c(15), yint, params)
494
495 ! 16th stage
496 aijkj = k(:,1)*dop853_a(106) + k(:,6)*dop853_a(111) + k(:,7)*dop853_a(112) + k(:,8)*dop853_a(113) &
497 + k(:,9)*dop853_a(114) + k(:,13)*dop853_a(118) + k(:,14)*dop853_a(119) + k(:,15)*dop853_a(120)
498 yint = y0 + h*aijkj
499 k(:,16) = pdiffeqsys%F(x0 + h*dop853_c(16), yint, params)
500
501 fcalls = 3
502
503 ! DOP853 dense output coefficients (see Hairer's code)
504
510
511 cont0(:) = y0(interpstates)
512
513 cont1(:) = y1(interpstates) - y0(interpstates)
514
515 cont2(:) = h*k(interpstates,1) - cont1(:)
516
517 cont3(:) = cont1(:) - h*k(interpstates,13) - cont2(:)
518
519 cont4(:) = h*(dop853_d(1,4)*k(interpstates,1) + dop853_d(6-4,4)*k(interpstates,6) &
520 + dop853_d(7-4,4)*k(interpstates,7) + dop853_d(8-4,4)*k(interpstates,8) &
521 + dop853_d(9-4,4)*k(interpstates,9) + dop853_d(10-4,4)*k(interpstates,10) &
522 + dop853_d(11-4,4)*k(interpstates,11) + dop853_d(12-4,4)*k(interpstates,12) &
523 + dop853_d(13-4,4)*k(interpstates,13) + dop853_d(14-4,4)*k(interpstates,14) &
524 + dop853_d(15-4,4)*k(interpstates,15) + dop853_d(16-4,4)*k(interpstates,16))
525
526 cont5(:) = h*(dop853_d(1,5)*k(interpstates,1) + dop853_d(6-4,5)*k(interpstates,6) &
527 + dop853_d(7-4,5)*k(interpstates,7) + dop853_d(8-4,5)*k(interpstates,8) &
528 + dop853_d(9-4,5)*k(interpstates,9) + dop853_d(10-4,5)*k(interpstates,10) &
529 + dop853_d(11-4,5)*k(interpstates,11) + dop853_d(12-4,5)*k(interpstates,12) &
530 + dop853_d(13-4,5)*k(interpstates,13) + dop853_d(14-4,5)*k(interpstates,14) &
531 + dop853_d(15-4,5)*k(interpstates,15) + dop853_d(16-4,5)*k(interpstates,16))
532
533 cont6(:) = h*(dop853_d(1,6)*k(interpstates,1) + dop853_d(6-4,6)*k(interpstates,6) &
534 + dop853_d(7-4,6)*k(interpstates,7) + dop853_d(8-4,6)*k(interpstates,8) &
535 + dop853_d(9-4,6)*k(interpstates,9) + dop853_d(10-4,6)*k(interpstates,10) &
536 + dop853_d(11-4,6)*k(interpstates,11) + dop853_d(12-4,6)*k(interpstates,12) &
537 + dop853_d(13-4,6)*k(interpstates,13) + dop853_d(14-4,6)*k(interpstates,14) &
538 + dop853_d(15-4,6)*k(interpstates,15) + dop853_d(16-4,6)*k(interpstates,16))

541 + dop853_d(7-4,7)*k(interpstates,7) + dop853_d(8-4,7)*k(interpstates,8) &

561 subroutine dop54_stepint(pDiffEqSys, X0, Y0, F0, h, k, Y1, Yint12, FCalls, Err, &
562 IsScalarTol, RTol, ATol, params)
563 class(DiffEqSys), intent(inout) :: pDiffEqSys
564 real(WP), intent(in) :: X0
565 real(WP), dimension(pDiffEqSys%n), intent(in) :: Y0
566 real(WP), dimension(pDiffEqSys%n), intent(inout) :: F0
567 real(WP), intent(in) :: h
568 real(WP), dimension(size(Y0),1:DOP54_sint), intent(inout) :: k
569 real(WP), dimension(size(Y0)), intent(out) :: Y1, Yint12
570 integer, intent(out) :: FCalls
571 real(WP), intent(out) :: Err
572 logical, intent(in), optional :: IsScalarTol
573 real(WP), dimension(size(Y0)), intent(in), optional :: RTol, ATol
574 real(WP), dimension(:), intent(in), optional :: params
575
576 real(WP), dimension(size(Y0)) :: Yint, Sc
577 logical :: EstimateErr
578
579 estimateerr = .false.
580 if (present(isscalartol) .AND. present(rtol) &
581 .AND. present(atol)) estimateerr = .true.
582
583 ! first stage (we expect F0 already contains the initial derivative)
584 k(:,1) = f0
585
586 ! 2nd stage
587 yint = y0 + h*(k(:,1)*dop54_a(1))
588 k(:,2) = pdiffeqsys%F(x0 + h*dop54_c(2), yint, params)
589
590 ! 3rd stage
591 yint = y0 + h*(k(:,1)*dop54_a(2) + k(:,2)*dop54_a(3))
592 k(:,3) = pdiffeqsys%F(x0 + h*dop54_c(3), yint, params)
593
594 ! 4th stage
595 yint = y0 + h*(k(:,1)*dop54_a(4) + k(:,2)*dop54_a(5) + k(:,3)*dop54_a(6))
596 k(:,4) = pdiffeqsys%F(x0 + h*dop54_c(4), yint, params)
597
598 ! 5th stage
599 yint = y0 + h*(k(:,1)*dop54_a(7) + k(:,2)*dop54_a(8) + k(:,3)*dop54_a(9) + k(:,4)*dop54_a(10))
600 k(:,5) = pdiffeqsys%F(x0 + h*dop54_c(5), yint, params)
601
602 ! 6th stage
603 yint12 = y0 + h*(k(:,1)*dop54_a(11) + k(:,2)*dop54_a(12) + k(:,3)*dop54_a(13) &
604 + k(:,4)*dop54_a(14) + k(:,5)*dop54_a(15))
605 k(:,6) = pdiffeqsys%F(x0 + h*dop54_c(6), yint12, params)
606
607 ! 7th stage
608 yint = y0 + h*(k(:,1)*dop54_a(16) + k(:,2)*dop54_a(17) + k(:,3)*dop54_a(18) &
609 + k(:,4)*dop54_a(19) + k(:,5)*dop54_a(20) + k(:,6)*dop54_a(21))
610 k(:,7) = pdiffeqsys%F(x0 + h, yint, params)
611
612 ! propagate the solution to the next step for error computation
613 ! For FSAL, Yint is the new solution
614
615 if (estimateerr) then
616 ! The scale factor for error computations
617 if (isscalartol .EQV. .false.) then
618 sc = atol + rtol*max(abs(y0), abs(yint))
619 else
620 sc = atol(1) + rtol(1)*max(abs(y0), abs(yint))
621 end if
622
623 ! Estimate the error. We assume that the error coeffcients are precomputed,
624 ! i.e., e_i = b_i - bhat_i
625 ! TBD: FSAL methods have this sum k(:,1)*e(1) + k(:,2)*e(2) + ... where the first and last
626 ! terms are same in magnitude but different in sign so can cause loss of precision
627 ! We may deivide e(n) by Sc first before multiplying with k and sum it
628 err = sqrt(sum(((h*(k(:,1)*dop54_e(1) + k(:,3)*dop54_e(3) + k(:,4)*dop54_e(4) &
629 + k(:,5)*dop54_e(5) + k(:,6)*dop54_e(6) + k(:,7)*dop54_e(7)))/sc)**2)/pdiffeqsys%n)
630 else
631 err = 0.0_wp
632 end if
633 !
634 ! If this step is accepted then return the new solution
635 ! else just return the initial condition
636 if (err <= 1.0_wp) then
637 y1 = yint
638 ! In FSAL methods, the intgeration first stage is same as the last stage
639 f0 = k(:,7)
640 ! Function calls made if the step is accepted
641 fcalls = dop54_sint - 1
642 else
643 y1 = y0
644 ! Function calls made if the step is rejected
645 fcalls = dop54_sint - 2
646 end if
647 end subroutine dop54_stepint
