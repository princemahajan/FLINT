69 subroutine root(a, b, ya, yb, tol, f, x, fx, niter, error)
70
71 implicit none
72
73 real(WP), intent(in) :: a
74 real(WP), intent(in) :: b
75 real(WP), intent(in) :: ya
76 real(WP), intent(in) :: yb
77 real(WP), intent(in) :: tol
78 procedure(func) :: f
79 real(WP), intent(out) :: x
80 real(WP), intent(out) :: fx
81 integer, intent(out) :: niter
82 integer, intent(out) :: error
83
84 real(WP) :: aa, bb, cc, fa, fb, fc, xm, ss, dd, ee, pp, qq, rr
85
86 real(WP) :: tolx
87
88 real(WP), parameter :: NEARZERO = 1.0e-20_wp
89
90 logical :: done
91 integer :: itr
92
93 aa = a
94 bb = b
95 fa = ya
96 fb = yb
97
98 error = 0
99 niter = 0
100
101 !! test
102 !fa = f(a)
103 !fb = f(b)
104
105 ! Trivial cases
106 if (abs(fa) <= nearzero) then
107 x = aa
108 fx = fa
109 return
110 else if (abs(fb) <= nearzero) then
111 x = bb
112 fx = fb
113 return
114 end if
115
116 ! if root is not bracketed then return error
117 if ((fa>0.0_wp .AND. fb>0.0_wp) .OR. (fa<0.0_wp .AND. fb<0.0_wp)) then
118 error = 1
119 niter = 0
120 return
121 end if
122
123 ! algorithm start here
124 fc = fb
125 done = .false.
126 itr = 0
127 do while (done .EQV. .false. .AND. itr < flint_rootfind_maxitr)
128
129 ! if root is NOT bracketed between C and B, then C = A
130 if ((fc>0.0_wp .AND. fb>0.0_wp) .OR. (fc<0.0_wp .AND. fb<0.0_wp)) then
131 cc = aa
132 fc = fa
133 dd = bb - aa
134 ee = dd
135 end if
136 ! Make sure |f(c)| is bigger than |f(b)|
137 if (abs(fc) < abs(fb)) then
138 aa = bb
139 bb = cc
140 cc = aa
141 fa = fb
142 fb = fc
143 fc = fa
144 end if
145 ! tolerance on the independent variable
146 tolx = 2.0_wp*flint_eps*abs(bb) + 0.5_wp*tol
147 ! middle point
148 xm = 0.5_wp*(cc - bb)
149
150 if (abs(xm) <= tolx .OR. abs(fa) < nearzero) then
151 ! root has been found
152 x = bb
153 done = .true.
154 fx = f(x)
155 else
156 if (abs(ee) >= tolx .AND. abs(fa) > abs(fb)) then
157 ss = fb/fa ! make sure |ss| < 1
158 if (abs(aa - cc) < nearzero) then
159 ! linear interpolation
160 pp = 2.0_wp*xm*ss
161 qq = 1.0_wp - ss
162 else
163 ! Inverse quadratic interpolation
164 qq = fa/fc
165 rr = fb/fc
166 pp = ss*(2.0_wp*xm*qq*(qq-rr)-(bb-aa)*(rr-1.0_wp))
167 qq = (qq - 1.0_wp)*(rr - 1.0_wp)*(ss - 1.0_wp)
168 end if
169
170 if (pp > nearzero) qq = -qq
171 pp = abs(pp)
172 if ((2.0_wp*pp) < min(3.0_wp*xm*qq-abs(tolx*qq), abs(ee*qq))) then
173 ! use linear or inverse quadratic interpolation
174 ee = dd
175 dd = pp/qq
176 else
177 ! use bisection
178 dd = xm
179 ee = dd
180 end if
181 else
182 ! use bisection
183 dd = xm
184 ee = dd
185 end if
186
187 aa = bb
188 fa = fb
189 if (abs(dd) > tolx) then
190 bb = bb + dd
191 else
192 ! Correction term too small, advance by at least tolx
193 if (xm > 0.0_wp) then
194 bb = bb + abs(tolx)
195 else
196 bb = bb - abs(tolx)
197 end if
198 end if
199 fb = f(bb)
200 itr = itr + 1
201 end if
202 end do
203
204 if (itr >= flint_rootfind_maxitr) error = 2
205 niter = itr
206
207 end subroutine root

214 pure function binarysearch(SortedX, SortedDataArray)
215
216 !import :: WP
217
218 implicit none
219
220 real(wp), dimension(1:), intent(in) :: sortedx
221 real(wp), dimension(:), intent(in) :: sorteddataarray
222 integer, dimension(size(SortedX)) :: binarysearch
223
224 integer :: lb, ub, a, b, m, n, ctr, xloc
225 real(wp) :: x
226
227 ! initialize
228 lb = 1
229 ub = size(sorteddataarray)
230 m = lb
231
232 ! for each element
233 do ctr = 1, size(sortedx)
234
235 ! element to search for
236 x = sortedx(ctr)
237
238 ! current interval. Note we assume that the array of the elements
239 ! to search for is also a sorted array.
240 a = m
241 b = ub
242
243 ! loop for finding the element
244 do
245 ! size of the current interval
246 n = b - a + 1
247
248 ! mid-point
249 m = floor((n+1)/2.0_wp) + (a-1)
250
251 if (x < sorteddataarray(m)) then
252 ! search the lower half
253 b = m
254 cycle
255 elseif (x > sorteddataarray(m) .AND. (m+1) /= b) then
256 ! search the upper half
257 a = m
258 cycle
259 else
260 ! We either have found the exact match
261 ! Or x lies in between Data(m) and Data(m+1)
262 xloc = m
263 exit
264 end if
265 end do
266
267 ! save the location of the current element
268 binarysearch(ctr) = xloc
269 end do
270
271 end function binarysearch
