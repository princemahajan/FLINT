29  module subroutine erk_int(me, x0, y0, xf, yf, stepsz, intstepson, xint, yint, eventmask, eventstates, stifftest, params)
30 
31  implicit none
32 
33  class(ERK_class), intent(inout) :: me
34 
35  real(WP), intent(in) :: X0
36  real(WP), dimension(me%pDiffEqSys%n), intent(in) :: Y0
37  real(WP), intent(inout) :: Xf
38  real(WP), dimension(me%pDiffEqSys%n), intent(out) :: Yf
39  real(WP), intent(inout) :: StepSz
40  logical, intent(in), optional :: IntStepsOn
41  real(WP), allocatable, dimension(:), intent(out), optional :: Xint
42  real(WP), allocatable, dimension(:,:), intent(out), optional :: Yint
43  logical, dimension(me%pDiffEqSys%m), intent(in), optional :: EventMask
44  real(WP), allocatable, dimension(:,:), intent(out), optional :: EventStates
45  integer, intent(inout), optional :: StiffTest
46  real(WP), dimension(:), intent(in), optional :: params
47 
48  real(WP) :: h, hSign, hnew, hmax, X, Err, Eventh, EventX0, EventX1, EventStepSz
49  real(WP), dimension(me%pDiffEqSys%n) :: Y1, Y2, F0, Sc0, Yint12
50 
51  !integer, dimension(:), allocatable :: IpStages
52  real(WP), dimension(size(me%InterpStates), 0:me%pstar) :: Bip
53 
54  real(WP), dimension(me%pDiffEqSys%m) :: EV0, EV1
55  integer, dimension(me%pDiffEqSys%m) :: EventDir
56  logical, dimension(me%pDiffEqSys%m) :: EventTerm
57  logical, dimension(me%pDiffEqSys%m) :: EvMask
58  real(WP), allocatable, dimension(:) :: EventData
59  real(WP), dimension(me%pDiffEqSys%n) :: EventY1, EventYm
60  real(WP), dimension(me%pDiffEqSys%n, me%s) :: k
61  real(WP), dimension(5) :: StepSzParams
62 
63  integer :: n, m, FCalls, TotalFCalls, AcceptedSteps, RejectedSteps
64  integer :: TotalSteps, MaxSteps, s, sint, p, pstar, q, EventId
65  integer :: StiffnessTest, StiffThreshold, NonStiffThreshold, StiffTestSteps
66 
67  logical :: IsProblemStiff, IntStepsNeeded, LastStepRejected, LAST_STEP
68  logical :: IsScalarTol, IsFSALMethod, InterpOn, EventsOn, BipComputed
69 
70  class(DiffEqSys), pointer :: pDiffEqSys
71  integer(kind(ERK_DOP853)) :: method
72 
73  integer(kind(FLINT_SUCCESS)) :: status, stat
74 
75  ! Init for this procedure
76  intstepsneeded = .false.
77  laststeprejected = .false.
78  last_step = .false.
79  status = flint_success
80 
81  ! Copy frequently needed scalars on the stack
82  n = me%pDiffEqSys%n
83  m = me%pDiffEqSys%m
84  p = me%p
85  pstar = me%pstar
86  q = me%q
87  pdiffeqsys => me%pDiffEqSys
88  maxsteps = me%MaxSteps
89  hmax = me%MaxStepSize
90  stepszparams = me%StepSzParams
91  method = me%method
92  isscalartol = me%IsScalarTol
93  isfsalmethod = me%IsFSALMethod
94  interpon = me%InterpOn
95  eventson = me%EventsOn
96  eventstepsz = me%EventStepSz
97  s = me%s
98  sint = me%sint
99  stiffteststeps = stifftest_steps
100 
101  ! If Interpolation is OFF and solution at integrator steps is needed,
102  ! then allocate the storage for steps that will be returned to the user.
103  ! If Interpolation is ON, then no need as we already allocated the internal
104  ! storage for the solution and we just use allocatable array assignment
105  if (present(intstepson)) then
106  if (intstepson .AND. interpon) then
107  status = flint_error_params
108  else if (intstepson .AND. (.NOT. interpon)) then
109  intstepsneeded = .true.
110  end if
111  end if
112 
113  if (intstepsneeded) then
114  allocate(xint(maxsteps), stat=stat)
115  if (stat /= 0) status = flint_error_memalloc
116  allocate(yint(pdiffeqsys%n,maxsteps), stat=stat)
117  if (stat /= 0) status = flint_error_memalloc
118  end if
119 
120  ! stiffness test options
121  if (present(stifftest)) then
122  stiffnesstest = stifftest
123  select case (stiffnesstest)
124  case (0)
125  case (1)
126  case (2)
127  case default
128  status = flint_error_params
129  end select
130  else
131  stiffnesstest = 0
132  end if
133 
134  ! make sure memory is allocated if interpolation is on
135  if (interpon) then
136  if (allocated(me%Xint) .EQV. .false. .OR. allocated(me%Bip) .EQV. .false.) then
137  me%status = flint_error_init_reqd
138  end if
139  ! If size of Xint is less than the max size, then it means user is calling
140  ! Intgerate again without doing Init first, then reallocate the memory
141  if (allocated(me%Xint)) then
142  if (size(me%Xint) < maxsteps) then
143  deallocate(me%Xint, stat = status)
144  deallocate(me%Bip, stat = status)
145  allocate(me%Xint(me%MaxSteps), stat=status)
146  allocate(me%Bip(size(me%InterpStates), 0:me%pstar, me%MaxSteps), stat=status)
147  if (status /= 0) me%status = flint_error_memalloc
148  end if
149  end if
150  end if
151 
152  ! If Mask is not specified, then all events are enabled by default
153  if (present(eventmask)) then
154  evmask = eventmask
155  else
156  evmask = .true.
157  end if
158 
159  ! If all the event masks are false then turn off events for this integration
160  if (.NOT. any(evmask)) eventson = .false.
161 
162  ! Make sure optional event arguments are present if events are ON.
163  if (eventson) then
164  if (.NOT. present(eventstates)) then
165  me%status = flint_error_eventparams
166  end if
167  ! allocate EventData to 0 size to define it
168  allocate(eventdata(0))
169  end if
170 
171  ! Return if any error up to this point
172  if (status /= flint_success) then
173  me%status = status
174  return
175  end if
176 
177  ! All the parameter checking should be before this point
178 
179  ! save the initial condition
180  me%X0 = x0
181  me%Y0 = y0
182 
183  ! compute the derivative and scale factor at the initial condition
184  f0 = pdiffeqsys%F(x0, y0, params)
185  totalfcalls = 1
186 
187  !! \remark Scale factor is computed using
188  !! \f[ SC_i = atol_i + rtol_i |Y_{i}| \quad i=1..n,\f]
189  !! where \f$|Y|\f$ is the absolute value of the provided vector
190  if (isscalartol .EQV. .false.) then
191  sc0 = me%ATol + me%RTol*abs(y0)
192  else
193  sc0 = me%ATol(1) + me%RTol(1)*abs(y0)
194  end if
195 
196  ! Determine the sign of the step: negative for backward integration
197  hsign = sign(1.0_wp, (xf-x0))
198 
199  ! Select the max step size if not provided by the user
200  if (hmax == 0.0_wp) hmax = abs(xf - x0)
201 
202  ! The initial step size: compute if user has not provided a non-zero value
203  if (stepsz == 0.0_wp) then
204  call stepsz0hairer(h, p, n, x0, y0, f0, sc0, hsign, hmax, fcalls, pdiffeqsys, params)
205  totalfcalls = totalfcalls + fcalls
206  else
207  h = stepsz
208  end if
209 
210  ! copy the initial condition as the first state
211  if (intstepsneeded) then
212  xint(1) = x0
213  yint(:,1) = y0
214  end if
215  if (interpon) then
216  me%Xint(1) = x0
217  !me%Yint(:,1) = Y0
218  end if
219 
220  ! Initialize the number of steps taken and other counters
221  x = x0
222  y1 = y0
223  totalsteps = 0
224  acceptedsteps = 0
225  rejectedsteps = 0
226  stiffthreshold = 0
227  nonstiffthreshold = 0
228  status = flint_success
229 
230  ! Main loop for computing the steps
231  do
232 
233  if (status == flint_success .AND. (.NOT. last_step) &
234  .AND. (totalsteps <= maxsteps)) then
235 
236  ! Adjust the step size if (X+h) is bigger than Xf
237  if (hsign*(x + 1.01_wp*h - xf) > 0.0) then
238  h = xf - x
239  last_step = .true.
240  end if
241 
242  ! Advance one step using the chosen method
243  if (method == erk_dop853) then
244  call dop853_stepint(x, y1, y2)
245  elseif (method == erk_dop54) then
246  call dop54_stepint(x, y1, y2)
247  else
248  call stepint(x, y1, y2)
249  end if
250 
251  ! update steps taken and function calls made
252  totalsteps = totalsteps + 1
253  totalfcalls = totalfcalls + fcalls
254 
255  ! check error to see if the step needs to be accepted or rejected
256  if (err <= 1.0_wp) then
257 
258  ! step is accepted
259  acceptedsteps = acceptedsteps + 1
260 
261  ! save the first accepted step-size
262  if (acceptedsteps == 1) me%h0 = h
263 
264  ! do we need stiffness detection?
265  if (stiffnesstest == 1 .OR. stiffnesstest == 2) then
266  call checkstiffness(isproblemstiff)
267  if (isproblemstiff) then
268  stifftest = -1
269  if (stiffnesstest == 1) then
270  status = flint_stiff_problem
271  last_step = .true.
272  end if
273  end if
274  end if
275 
276  ! Events checking start here, ignore event checking at IC
277  bipcomputed = .false.
278  if (eventson .AND. acceptedsteps > 1) then
279  ! generate X values at which event triggers are checked
280  if (eventstepsz == 0.0 .OR. eventstepsz >= abs(h)) then
281  eventh = h
282  else
283  eventh = sign(eventstepsz, h)
284  ! we need interpolation coefficinets to check signs at the intermediate points
285  if (.NOT. bipcomputed) then
286  select case (method)
287  case (erk_dop853)
288  bip = dop853_intpcoeff(x, y1, x+h, y2, me%InterpStates)
289  case (erk_dop54)
290  bip = dop54_intpcoeff(x, y1, x+h, y2, me%InterpStates)
291  case default
292  bip = intpcoeff(x, y1, x+h, y2, me%InterpStates)
293  end select
294  totalfcalls = totalfcalls + fcalls
295  bipcomputed = .true.
296  end if
297  end if
298 
299  eventx0 = x
300  eventx1 = x + eventh
301  ! The event value at the first point where event is checked
302  if (acceptedsteps == 2) call pdiffeqsys%G(0, x, y1, ev0, eventdir, eventterm)
303 
304  ! check for event triggers between X and X+h
305  eventloop: do
306  block
307  logical :: EventTriggered
308  real(WP) :: val0, val1, valm, EventXm
309  integer :: rootiter, rooterror
310 
311  !! interpolate solution
312  if (eventh == h) then
313  eventy1 = y2
314  else
315  eventy1 = interpy(size(me%InterpStates), eventx1, x, h, me%pstar, bip)
316  end if
317  ! evaluate all event functions
318  call pdiffeqsys%G(0, eventx1, eventy1, ev1, eventdir, eventterm)
319 
320  ! check each event for trigger
321  do eventid = 1,m
322  ! if event is masked, no event check
323  if (evmask(eventid) .EQV. .false.) cycle
324 
325  val0 = ev0(eventid)
326  val1 = ev1(eventid)
327 
328  ! If the previous or the new value is NaN, ignore the event
329  if (ieee_is_nan(val0) .OR. ieee_is_nan(val1)) cycle
330 
331  eventtriggered = .false.
332  select case (eventdir(eventid))
333  case (0)
334  if ((val0<=0 .AND. val1>=0) &
335  .OR.(val0>=0 .AND. val1<=0)) eventtriggered = .true.
336  case (1)
337  if ((val0<=0 .AND. val1>=0)) eventtriggered = .true.
338  case (-1)
339  if ((val0>=0 .AND. val1<=0)) eventtriggered = .true.
340  end select
341 
342  if (eventtriggered) then
343  ! Current event has indeed triggered
344  ! compute interpolation coefficients if not already computed
345  if (.NOT. bipcomputed) then
346  select case (method)
347  case (erk_dop853)
348  bip = dop853_intpcoeff(x, y1, x+h, y2, me%InterpStates)
349  case (erk_dop54)
350  bip = dop54_intpcoeff(x, y1, x+h, y2, me%InterpStates)
351  case default
352  bip = intpcoeff(x, y1, x+h, y2, me%InterpStates)
353  end select
354  totalfcalls = totalfcalls + fcalls
355  bipcomputed = .true.
356  end if
357  ! Now find the exact event location using root-finding
358  call root(eventx0, eventx1, val0, val1, default_eventtol, &
359  singleevent, eventxm, valm, rootiter, rooterror)
360  if (rooterror == 0 .OR. rooterror == 2) then
361  ! save the event states
362  eventym = interpy(size(me%InterpStates), eventxm, x, h, me%pstar, bip)
363  eventdata = [eventdata, eventxm, eventym, real(eventid, wp)]
364  else
365  ! root finding failed, save the event states at the point "a"
366  status = flint_error_eventroot
367  end if
368  ! Do we need to stop at this event?
369  if (eventterm(eventid) .EQV. .true.) then
370  ! the last solution is the same as the event state
371  h = eventxm - x
372  y2 = eventym
373  last_step = .true.
374  status = flint_event_term
375  exit eventloop
376  end if
377  end if
378  end do
379 
380  ! new check point for event
381  eventx0 = eventx1
382  ev0 = ev1
383  if (eventx1 == x + h) exit eventloop ! we are done
384  if (eventx1 + eventh > x + h) then
385  eventx1 = x + h
386  else
387  eventx1 = eventx1 + eventh
388  end if
389  end block
390  end do eventloop
391  end if
392 
393  ! return the solution at integrator's natural step size
394  if (intstepsneeded) then
395  xint(acceptedsteps + 1) = x + h
396  yint(:,acceptedsteps + 1) = y2
397  end if
398 
399  ! Do we need to store interpolation coefficients?
400  if (interpon) then
401  if (.NOT. bipcomputed) then
402  ! Compute interpolation coefficients
403  select case (method)
404  case (erk_dop853)
405  bip = dop853_intpcoeff(x, y1, x+h, y2, me%InterpStates)
406  case (erk_dop54)
407  bip = dop54_intpcoeff(x, y1, x+h, y2, me%InterpStates)
408  case default
409  bip = intpcoeff(x, y1, x+h, y2, me%InterpStates)
410  end select
411  bipcomputed = .true.
412  totalfcalls = totalfcalls + fcalls
413  end if
414  ! store the solution internally at the natural step size
415  me%Xint(acceptedsteps + 1) = x + h
416  !me%Yint(:,AcceptedSteps + 1) = Y2
417  me%Bip(:,:,acceptedsteps) = bip
418  end if
419 
420  ! get ready for the next iteration
421  x = x + h
422  y1 = y2
423 
424  ! compute the new step size now
425  hnew = stepszhairer(h, .false., q, err, stepszparams)
426 
427  ! make sure hnew is not greater than hmax
428  if (abs(hnew) > hmax) hnew = hsign*hmax
429 
430  ! if the last step was rejected, then dont increase step size
431  if (laststeprejected .EQV. .true.) hnew = hsign*min(abs(hnew),abs(h))
432 
433  ! Update the Lund stabilization parameter hbyhoptOld
434  ! this should be after computing the new step size
435  stepszparams(5) = max(err, hbyhoptold)
436 
437  ! reset
438  laststeprejected = .false.
439  else
440 
441  ! step is rejected, now what
442 
443  ! increment the reject step counter, ignore the initial step rejections
444  if (acceptedsteps >= 1) rejectedsteps = rejectedsteps + 1
445 
446  ! compute the new step size now
447  hnew = stepszhairer(h, .true., q, err, stepszparams)
448 
449  ! Make note of your rejection. It is used in the next iteration
450  ! for computing the new step size.
451  laststeprejected = .true.
452 
453  ! Retry again if this was the last step
454  last_step = .false.
455 
456  end if
457 
458  ! update the step size for next iteration
459  h = hnew
460 
461  ! Take another step only if step size is not too small
462  ! Reference: Hairer's DOP853 code
463  if (abs(h)*0.01_wp <= abs(x)*eps) then
464  status = flint_error_stepsz_toosmall
465  end if
466 
467  else
468 
469  ! We have either accomplished our job or finished prematurely?!
470 
471  ! return the solution at the integrator's natural steps
472  if (intstepsneeded) then
473  block
474  real(WP), allocatable, dimension(:) :: Xarr
475  real(WP), allocatable, dimension(:,:) :: Yarr
476 
477  allocate(xarr(acceptedsteps + 1))
478  allocate(yarr(n,acceptedsteps + 1))
479 
480  xarr = xint(1:acceptedsteps + 1)
481  yarr = yint(:,1:acceptedsteps + 1)
482 
483  call move_alloc(from=xarr, to=xint)
484  call move_alloc(from=yarr, to=yint)
485  end block
486  end if
487 
488  ! Update internal storage size based on accepted steps
489  if (interpon) then
490  block
491  real(WP), allocatable, dimension(:) :: Xarr
492  !real(WP), allocatable, dimension(:,:) :: Yarr
493  real(WP), allocatable, dimension(:,:,:) :: Biparr
494 
495  allocate(xarr(acceptedsteps + 1))
496  !allocate(Yarr(n,AcceptedSteps + 1))
497  allocate(biparr(size(me%InterpStates),0:me%pstar,acceptedsteps))
498 
499  xarr = me%Xint(1:acceptedsteps + 1)
500  !Yarr = me%Yint(:,1:AcceptedSteps + 1)
501  biparr = me%Bip(:,:,1:acceptedsteps)
502 
503  call move_alloc(from=xarr, to=me%Xint)
504  !call move_alloc(from=Yarr, to=me%Yint)
505  call move_alloc(from=biparr, to=me%Bip)
506  end block
507  end if
508 
509  ! Prepare event output data
510  if (eventson .AND. allocated(eventdata)) then
511  eventstates = reshape(eventdata, [n+2, int(size(eventdata)/(n+2))])
512  end if
513 
514  if (x == xf .AND. status == flint_success) then
515  ! we finished like we should
516  elseif (eventson .AND. status == flint_event_term) then
517  ! One of the terminal event has triggered, do nothing
518  else if (totalsteps > maxsteps .AND. status == flint_success) then
519  ! maximum steps reached
520  status = flint_error_maxsteps
521  else if (status == flint_error_stepsz_toosmall) then
522  ! do nothing
523  else if (eventson .AND. status == flint_error_eventroot) then
524  ! do nothing
525  else if (status == flint_stiff_problem) then
526  ! do nothing
527  else
528  ! Some unknown error (We should never be here!)
529  status = flint_error
530  end if
531 
532  ! save the stats
533  xf = x
534  yf = y1
535  stepsz = h
536  me%Xf = xf
537  me%hf = stepsz
538  me%Yf = yf
539  me%AcceptedSteps = acceptedsteps
540  me%RejectedSteps = rejectedsteps
541  me%FCalls = totalfcalls
542  me%status = status
543 
544  ! exit the main integration loop
545  exit
546 
547  end if
548 
549  end do

557  subroutine stepint(X0, Y0, Y1)
558 
559  implicit none
560 
561  real(WP), intent(in) :: X0
562  real(WP), dimension(n), intent(in) :: Y0
563  real(WP), dimension(n), intent(out) :: Y1
564 
565  real(WP), dimension(n) :: Yint, Sc, aijkj
566  integer :: i
567 
568  ! first stage
569  k(:,1) = f0
570 
571  ! compute rest of the stages
572  ! This compact and beautiful code is slower than ugly hardcoded one!
573  do i = 2,sint
574  block
575  integer :: astart, j
576  ! starting index of a_ij, where i=i, j=1:(i-1)
577  ! It is a series sum: 1 + 2 + 3 + ...
578  astart = int((i-1)/2.0*(i-2))
579  ! In my testing, I am seeing matmul performing better than
580  ! MKL's gemv. However, do loop is faster than both but
581  ! but still slower than hard-coded computations
582  ! call gemv(k(:,1:(i-1)), me%a((astart+1):(i-1)), aijkj)
583  !aijkj = matmul(k(:,1:(i-1)), me%a((astart+1):(astart+i-1)))
584  aijkj = 0.0_wp
585  do j = 1,(i-1)
586  aijkj = aijkj + k(:,j)*me%a(astart + j)
587  end do
588 
589  yint = y0 + h*aijkj
590  k(:,i) = pdiffeqsys%F(x0 + h*me%c(i), yint, params)
591 
592  ! Yint for the last non-FSAL stage is needed for stiffness detection
593  ! This is from Hairer's DOP853, so not sure its correct for other solvers
594  if (i == (sint-1)) yint12 = yint
595  end block
596  end do
597 
598  ! propagate the solution to the next step for error computation
599  ! For FSAL methods, Yint already has the new solution
600 
601  ! For non-FSAL methods, compute the solution at the next step
602  if (isfsalmethod .EQV. .false.) then
603  ! for stiffness detection, we need last non-FSAL stage solution
604  yint12 = yint
605  yint = y0 + h*matmul(k(:,1:sint), me%b(1:sint))
606  end if
607 
608  ! The scale factor for error computations
609  if (isscalartol .EQV. .false.) then
610  sc = me%ATol + me%RTol*max(abs(y0), abs(yint))
611  else
612  sc = me%ATol(1) + me%RTol(1)*max(abs(y0), abs(yint))
613  end if
614 
615  ! Estimate the error. We assume that the error coeffcients are precomputed,
616  ! i.e., e_i = b_i - bhat_i
617  block
618  real(WP), dimension(n) :: temp
619  integer :: j
620  temp = 0.0_wp
621  do concurrent(j = 1:sint)
622  temp = temp + k(:,j)*me%e(j)
623  end do
624  err = abs(h)*sqrt(sum((temp/sc)**2)/n)
625  end block
626 
627  ! If this step is accepted then return the new solution
628  ! else just return the initial condition
629  if (err <= 1.0_wp) then
630  y1 = yint
631  if (isfsalmethod .EQV. .false.) then
632  ! Evaluate F0 for the next step
633  f0 = pdiffeqsys%F(x0 + h, y1, params)
634  ! Number of function calls made so far
635  fcalls = sint
636  else
637  ! In FSAL methods, the intgeration last stage is F0 for the next step
638  f0 = k(:,sint)
639  ! Function calls made if the step is accepted
640  fcalls = sint - 1
641  end if
642  else
643  y1 = y0
644  ! Function calls made if the step is rejected
645  if (isfsalmethod .EQV. .false.) then
646  fcalls = sint - 1
647  else
648  fcalls = sint - 2
649  end if
650  end if
