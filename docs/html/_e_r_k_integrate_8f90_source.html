29 ! TBD: add support for taking a single step
30 module subroutine erk_int(me, x0, y0, xf, yf, stepsz, useconststepsz, intstepson, &
31 xint, yint, eventmask, eventstates, stifftest, params)
32
33 implicit none
34
35 class(ERK_class), intent(inout) :: me
36
37 real(WP), intent(in) :: X0
38 real(WP), dimension(me%pDiffEqSys%n), intent(in) :: Y0
39 real(WP), intent(inout) :: Xf
40 real(WP), dimension(size(Y0)), intent(out) :: Yf
41 real(WP), intent(inout) :: StepSz
42 logical, intent(in), optional :: UseConstStepSz
43 logical, intent(in), optional :: IntStepsOn
44 real(WP), allocatable, dimension(:), intent(out), optional :: Xint
45 real(WP), allocatable, dimension(:,:), intent(out), optional :: Yint
46 logical, dimension(me%pDiffEqSys%m), intent(in), optional :: EventMask
47 real(WP), allocatable, dimension(:,:), intent(out), optional :: EventStates
48 integer, intent(inout), optional :: StiffTest
49 real(WP), dimension(:), intent(in), optional :: params
50
51
52 class(DiffEqSys), pointer :: pDiffEqSys
53
54 integer(kind(FLINT_SUCCESS)) :: status, stat
55
56 real(WP), dimension(size(me%InterpStates), 0:me%pstar) :: Bip
57 real(WP), dimension(6) :: StepSzParams
58
59 real(WP) :: h, hSign, hnew, hmax, X, Err, LastStepSzFac
60 real(WP), dimension(me%pDiffEqSys%n) :: Y1, Y2, F0, Sc0, Yint12
61
62 integer :: FCalls, nInterpStates
63 integer :: StiffnessTest, StiffThreshold, NonStiffThreshold, StiffTestSteps
64
65 logical :: IntStepsNeeded, LastStepRejected, LAST_STEP, ConstStepSz, IsProblemStiff
66 logical :: InterpOn, EventsOn, BipComputed
67
68 ! Events related variables
69 integer :: EventId
70 real(WP) :: EventX0, EventX1
71 real(WP), dimension(me%pDiffEqSys%n) :: EventNewY
72 integer, dimension(me%pDiffEqSys%m) :: EvalEvents, StepSzEvents, EventDir, nStepSzEvents
73 logical, dimension(me%pDiffEqSys%m) :: EvMask, EventsDetected
74 integer(kind(FLINT_EVENTACTION_CONTINUE)) :: EventAction
75 real(WP), dimension(MAXNUMSTEPSZEVENTS, me%pDiffEqSys%m) :: EX0, EX1
76 real(WP), dimension(MAXNUMSTEPSZEVENTS, me%pDiffEqSys%m) :: EV0, EV1
77 real(WP), dimension(:), allocatable :: EventData
78
79 ! Init for this procedure
80 intstepsneeded = .false.
81 laststeprejected = .false.
82 last_step = .false.
83 status = flint_success
84
85 pdiffeqsys => me%pDiffEqSys
86 hmax = me%MaxStepSize
87 interpon = me%InterpOn
88 eventson = me%EventsOn
89 ninterpstates = size(me%InterpStates)
90 stepszparams = me%StepSzParams
91 stiffteststeps = stifftest_steps
92
93 ! Sign of the step: negative for backward integration
94 hsign = sign(1.0_wp, (xf-x0))
95
96 ! If UseConstStepSz option is present, then use it. Adaptive step size
97 ! algorithm is used by default. If constant step size is selected then
98 ! StepSz must have proper value set by the user along with its sign.
99 conststepsz = .false.
100 if (present(useconststepsz)) then
101 conststepsz = useconststepsz
102 if (conststepsz .AND. ((hsign*stepsz) <= 0)) status = flint_error_conststepsz
103 end if
104
105 ! set the last step size expansion factor
106 if (conststepsz) then
107 laststepszfac = 1.0_wp
108 else
109 laststepszfac = laststep_expfac
110 end if
111
112 ! If Interpolation is OFF and solution at integrator steps is needed,
113 ! then allocate the storage for steps that will be returned to the user.
114 ! If Interpolation is ON, then no need as we already allocated the internal
115 ! storage for the solution and we just use allocatable array assignment
116 if (present(intstepson)) then
117 if (intstepson .AND. interpon) then
118 ! Both options mustn't be True at the same time
119 status = flint_error_params
120 else if (intstepson .AND. (.NOT. interpon)) then
121 intstepsneeded = .true.
122 end if
123 end if
124
125 if (intstepsneeded) then
126 if (present(xint) .AND. present(yint)) then
127 ! allocate 1 more than max steps to accommodate the initial condition
128 allocate(xint(me%MaxSteps+1), stat=stat)
129 if (stat /= 0) status = flint_error_memalloc
130 allocate(yint(pdiffeqsys%n,me%MaxSteps+1), stat=stat)
131 if (stat /= 0) status = flint_error_memalloc
132 else
133 ! With IntStepsNeeded, Xint and Yint must be provided
134 status = flint_error_params
135 end if
136 end if
137
138 ! stiffness test options
139 if (present(stifftest)) then
140 stiffnesstest = stifftest
141 select case (stiffnesstest)
142 case (0)
143 case (1)
144 case (2)
145 case default
146 status = flint_error_params
147 end select
148 else
149 stiffnesstest = 0
150 end if
151
152 ! make sure memory is allocated if interpolation is on
153 if (interpon) then
154 if ((.NOT. allocated(me%Xint)) .OR. (.NOT. allocated(me%Bip))) then
155 me%status = flint_error_init_reqd
156 end if
157 ! If size of Xint is less than the max size, then it means user is calling
158 ! Intgerate again without doing Init first, so reallocate the memory
159 if (allocated(me%Xint)) then
160 if (size(me%Xint) < me%MaxSteps+1) then
161 deallocate(me%Xint, stat = status)
162 deallocate(me%Bip, stat = status)
163 allocate(me%Xint(me%MaxSteps+1), stat=status)
164 allocate(me%Bip(ninterpstates, 0:me%pstar, me%MaxSteps+1), stat=status)
165 if (status /= 0) me%status = flint_error_memalloc
166 end if
167 end if
168 end if
169
170 ! Make sure optional event arguments are present if events are enabled
171 if (eventson) then
172 ! If Mask is not specified, then all events are enabled by default
173 if (present(eventmask)) then
174 evmask = eventmask
175 else
176 evmask = .true.
177 end if
178
179 if (.NOT. any(evmask)) then
180 ! If all the event masks are false then turn off events for this integration
181 eventson = .false.
182 elseif (.NOT. present(eventstates)) then
183 ! If user has not provided the allocatable array for storage, retrun error
184 me%status = flint_error_eventparams
185 else
186 ! Allocate EventData to 0 size to define it
187 allocate(eventdata(0))
188 end if
189 end if
190
191 ! Return if any error up to this point
192 if (status /= flint_success) then
193 me%status = status
194 return
195 end if
196
197 ! All the parameter checking should be done before this point
198
199 ! Initialize the counters to prepare for the main integration loop
200 x = x0
201 y1 = y0
202 me%FCalls = 0
203 me%TotalSteps = 0
204 me%AcceptedSteps = 0
205 me%RejectedSteps = 0
206 stiffthreshold = 0
207 nonstiffthreshold = 0
208 status = flint_success
209
210 ! Evaluate the event function at the initial condition to check for
211 ! events between X0 and the first integrated state
212 if (eventson) then
213 eventx0 = x
214 eventnewy = y1
215 evalevents = 1
216 call pdiffeqsys%G(eventx0, eventnewy, evalevents, ev0(1,:), eventdir)
217 end if
218
219 ! Compute the derivative and scale factor at the initial condition
220 f0 = pdiffeqsys%F(x, y1, params)
221 me%FCalls = me%FCalls + 1
222
223 !! \remark Scale factor is computed using
224 !! \f[ SC_i = atol_i + rtol_i |Y_{i}| \quad i=1..n,\f]
225 !! where \f$|Y|\f$ is the absolute value of the provided vector
226 if (.NOT. me%IsScalarTol) then
227 sc0 = me%ATol + me%RTol*abs(y1)
228 else
229 sc0 = me%ATol(1) + me%RTol(1)*abs(y1)
230 end if
231
232 ! Select the max step size if not provided by the user
233 if (hmax == 0.0_wp) hmax = abs(xf - x)
234
235 ! The initial step size: compute if user has not provided a non-zero value
236 if (stepsz == 0.0_wp) then
237 call stepsz0hairer(h, me%p, pdiffeqsys%n, x, y1, f0, sc0, hsign, hmax, fcalls, pdiffeqsys, params)
238 me%FCalls = me%FCalls + fcalls
239 else
240 h = stepsz
241 end if
242
243 ! save the initial condition and the initial derivative
244 me%X0 = x
245 me%Y0 = y1
246 me%k(:,1) = f0
247
248 ! copy the initial condition as the first state
249
250 if (intstepsneeded) then
251 xint(1) = x
252 yint(:,1) = y1
253 end if
254
255 if (interpon) me%Xint(1) = x
256
257 ! Main loop for computing the steps
258
259 do while (status == flint_success .AND. (.NOT. last_step) &
260 .AND. (me%TotalSteps < me%MaxSteps))
261
262 ! if we are only slightly away from the final state
263 ! then adjust to step size to finish the integration
264 if (hsign*(x + laststepszfac*h - xf) > 0) then
265 h = xf - x
266 last_step = .true.
267 end if
268
269 ! Advance one step using the chosen method
270 call me%StepInt(x, y1, h, y2, yint12, fcalls, (.NOT. conststepsz), err, params)
271
272 ! update steps taken and function calls made
273 me%TotalSteps = me%TotalSteps + 1
274 me%FCalls = me%FCalls + fcalls
275
276 ! check error to see if the step needs to be accepted or rejected
277 ! For constant step size, Err must be set to 0 by stepint function
278
279 if (err <= 1.0_wp) then
280
281 ! step is accepted
282 me%AcceptedSteps = me%AcceptedSteps + 1
283
284 ! save the first accepted step-size
285 if (me%AcceptedSteps == 1) me%h0 = h
286
287 ! do we need stiffness detection?
288 if (stiffnesstest == 1 .OR. stiffnesstest == 2) then
289 call checkstiffness(isproblemstiff)
290 if (isproblemstiff) then
291 stifftest = -1
292 if (stiffnesstest == 1) then
293 status = flint_stiff_problem
294 last_step = .true.
295 end if
296 end if
297 end if
298
299 ! Event Detection start here after a step is accepted
300
301 ! reset the flag for computing interpolation coefficients
302 bipcomputed = .false.
303
304 if (eventson) then
305
306 evalevents = 1
307 eventsdetected = .false.
308 eventx0 = x
309 eventx1 = x + h
310 ex0 = eventx0
311 ex1 = eventx1
312 nstepszevents = 1
313
314 ! First check for events sign change at the step boundaries
315 eventnewy = y2
316 call pdiffeqsys%G(eventx1, eventnewy, evalevents, ev1(1,:), eventdir)
317 do eventid = 1,pdiffeqsys%m
318 call detectevent(eventid, ev0(1,eventid), ev1(1,eventid), eventsdetected(eventid))
319 end do
320
321 ! Check do we need to check sign changes for any event at smaller intervals
322 ! than the current step length
323 stepszevents = 1
324 where (me%EventStepSz == 0 .OR. me%EventStepSz > hsign*h)
325 stepszevents = 0
326 end where
327
328 if (any(stepszevents == 1)) then
329
330 ! For computing solution between the steps
331 if (.NOT. bipcomputed) then
332 call me%InterpCoeff(x, y1, h, y2, bip, fcalls, params)
333 me%FCalls = me%FCalls + fcalls
334 bipcomputed = .true.
335 end if
336
337 ! Check for sign changes one by one for these events
338 do eventid = 1, pdiffeqsys%m
339 block
340 real(WP), dimension(pDiffEqSys%m) :: EventV0, EventV1, EventVh
341 real(WP) :: Eventh
342 logical :: EvStepSzDetected, LastStepSzEvent
343
344 ! Skip the event if statically masked
345 if ((evmask(eventid)) .AND. &
346 (stepszevents(eventid)==1)) then
347
348 evalevents = 0
349 evalevents(eventid) = 1
350
351 eventh = hsign*me%EventStepSz(eventid)
352 eventx1 = x + eventh
353 eventv0 = ev0(1,:)
354 eventv1 = ev1(1,:)
355
356 ! Save the original event values at X+h
357 eventvh(eventid) = ev1(1,eventid)
358
359 eventsdetected(eventid) = .false.
360 laststepszevent = .false.
361
362 ! detect events inside the integrator's step
363 do while (eventx1 <= (x+h))
364 if (eventx1 /= (x+h)) then
365 ! interpolate solution
366 eventnewy = interpy(pdiffeqsys%n, eventx1, x, h, me%pstar, bip)
367 ! evaluate this specific event function
368 call pdiffeqsys%G(eventx1, eventnewy, evalevents, &
369 eventv1, eventdir)
370 else
371 eventv1(eventid) = eventvh(eventid)
372 end if
373 call detectevent(eventid, eventv0(eventid),eventv1(eventid),evstepszdetected)
374 if (evstepszdetected) then
375 ! store the this event details
376 eventsdetected(eventid) = .true.
377 ex0(nstepszevents(eventid),eventid) = eventx0
378 ex1(nstepszevents(eventid),eventid) = eventx1
379 ev0(nstepszevents(eventid),eventid) = eventv0(eventid)
380 ev1(nstepszevents(eventid),eventid) = eventv1(eventid)
381 nstepszevents(eventid) = nstepszevents(eventid) + 1
382 if (nstepszevents(eventid) > maxnumstepszevents) then
383 ! max number of step size events has been detected
384 exit
385 end if
386 end if
387 ! prepare for the next step interval
388 if (laststepszevent) exit
389 eventx0 = eventx1
390 eventx1 = eventx1 + eventh
391 if (eventx1 > x+h) then
392 eventx1 = x + h
393 laststepszevent = .true.
394 end if
395 eventv0(eventid) = eventv1(eventid)
396 end do
397 ! nStepSzEvents always counts one more than actual no. of
398 ! step size events, so we need to subtract 1
399 if (nstepszevents(eventid) > 1) nstepszevents(eventid) = nstepszevents(eventid) - 1
400 end if
401 end block
402 end do
403 end if
404
405 ! If any event is triggered?
406
407 if (any(eventsdetected)) then
408 block
409
410 real(WP), dimension(MAXNUMSTEPSZEVENTS,pDiffEqSys%m) :: EXm
411 real(WP), dimension(pDiffEqSys%n) :: EYm
412 integer :: StepEvId, rootiter, rooterror, nEvents
413 integer, dimension(1) :: MinEvXIndex
414
415 ! Locate the events first
416
417 do eventid = 1, pdiffeqsys%m
418 do stepevid = 1, nstepszevents(eventid)
419 if (eventsdetected(eventid)) then
420
421 select case (me%EventOptions(eventid))
422 ! Left bounrday of the integrator step is the event location
423 case (flint_eventoption_stepbegin)
424 exm(stepevid,eventid) = ex0(stepevid,eventid)
425
426 ! Right bounrday of the integrator step is the event location
427 case (flint_eventoption_stepend)
428 exm(stepevid,eventid) = ex1(stepevid,eventid)
429
430 case default
431 ! root-finding is enabled, so locate the event exactly
432 if (.NOT. bipcomputed) then
433 call me%InterpCoeff(x, y1, h, y2, bip, fcalls, params)
434 me%FCalls = me%FCalls + fcalls
435 bipcomputed = .true.
436 end if
437
438 call root(ex0(stepevid,eventid), ex1(stepevid, eventid), &
439 ev0(stepevid,eventid), ev1(stepevid,eventid), &
440 me%ETol(eventid), singleevent, &
441 exm(stepevid,eventid), ev0(stepevid,eventid), &
442 rootiter, rooterror)
443 if (rooterror /= 0 .AND. rooterror /= 2) then
444 ! root finding failed, dont exit. Try to go as far as you can!
445 status = flint_error_eventroot
446 end if
447 end select
448
449 end if
450 end do
451 end do
452
453 ! Handle the event action starting from the earliest event
454
455 ! iterate until all events are handled
456 do while (any(eventsdetected))
457
458 ! Find the earliest event. Note that the events' location
459 ! increases with the 1st index of EXm.
460 minevxindex = minloc(exm(1,:), eventsdetected)
461
462 ! solution at the event
463 select case (me%EventOptions(minevxindex(1)))
464 case (flint_eventoption_stepbegin)
465 eym = y1
466 case (flint_eventoption_stepend)
467 eym = y2
468 case default
469 eym = interpy(ninterpstates, exm(1,minevxindex(1)),&
470 x, h, me%pstar, bip)
471 end select
472
473 ! Call the event function with the located event info
474
475 evalevents = 0
476 evalevents(minevxindex(1)) = 1
477 eventnewy = eym
478 call pdiffeqsys%G(exm(1,minevxindex(1)), eventnewy, evalevents, &
479 ev0(1,:), eventdir, minevxindex(1), eventaction)
480
481 ! save the event state
482 ! Note that if an event action alters the solution Y, event data returned to
483 ! the user still contains the original state corresponding to the event location
484 eventdata = [eventdata, exm(1,minevxindex(1)), eym, real(minevxindex(1), wp)]
485
486 ! Handle the event actions. Note that we must recompute interpolation
487 ! coefficients in case we update the solution Y2 or h.
488
489 ! Handle the event mask action
490 if (btest(eventaction, 7)) then
491 evmask(minevxindex(1)) = .false.
492 end if
493
494 ! The following actions are mutually exclusive
495 select case (iand(eventaction, b'01111111'))
496 case (flint_eventaction_terminate)
497 ! Return the event solution as the last solution
498 h = exm(1,minevxindex(1)) - x
499 y2 = eym
500 last_step = .true.
501 bipcomputed = .false.
502 status = flint_event_term
503 exit
504 case (flint_eventaction_restartint)
505 ! Restart the integration from this point on
506 h = exm(1,minevxindex(1)) - x
507 y2 = eym
508 ! If it was already a last step, then reset it
509 last_step = .false.
510 bipcomputed = .false.
511 exit
512 case (flint_eventaction_changesoly)
513 ! Change the solution Y and restart the integration
514 h = exm(1,minevxindex(1)) - x
515 y2 = eym
516 ! If it was already a last step, then reset it
517 last_step = .false.
518 bipcomputed = .false.
519 exit
520 end select
521
522 ! This event has been handled, so remove it for the next event
523 ! If the handled event is a step-size event then we need to
524 ! shift the contents of its columns to left
525 if (stepszevents(minevxindex(1)) == 1) then
526 nstepszevents(minevxindex(1)) = nstepszevents(minevxindex(1)) - 1
527 if (nstepszevents(minevxindex(1)) == 0) then
528 eventsdetected(minevxindex(1)) = .false.
529 else
530 exm(1:nstepszevents(minevxindex(1)), minevxindex(1)) &
531 = exm(2:(nstepszevents(minevxindex(1))+1),minevxindex(1))
532 end if
533 else
534 eventsdetected(minevxindex(1)) = .false.
535 end if
536 end do
537 end block
538 end if
539
540 ! save EV0 for the next integrator's step
541 ev0 = ev1
542
543 end if
544
545 ! Do we need to store interpolation coefficients?
546 if (interpon) then
547 ! If EventAction is performed to terminate, restart or change SolY
548 ! then we must compute interpolation coefficients again.
549 if (.NOT. bipcomputed) then
550 call me%InterpCoeff(x, y1, h, y2, bip, fcalls, params)
551 me%FCalls = me%FCalls + fcalls
552 bipcomputed = .true.
553 end if
554 ! store the solution internally at the natural step size
555 me%Xint(me%AcceptedSteps + 1) = x + h
556 me%Bip(:,:,me%AcceptedSteps) = bip
557 end if
558
559
560 ! Restart the integration and update Y2 with the modified solution.
561 ! Note that the modified solution is returned back to the user in Yint.
562 ! The following code must be after the code for saving interpolation
563 ! coefficients as these coefficients need the original Y only.
564 if (eventson) then
565 select case (iand(eventaction, b'01111111'))
566 case(flint_eventaction_restartint)
567 me%k(:,1) = pdiffeqsys%F(x + h, y2, params)
568 me%FCalls = me%FCalls + 1
569 case(flint_eventaction_changesoly)
570 y2 = eventnewy
571 me%k(:,1) = pdiffeqsys%F(x + h, y2, params)
572 me%FCalls = me%FCalls + 1
573 end select
574 end if
575
576 ! return the solution at integrator's natural step size
577 if (intstepsneeded) then
578 xint(me%AcceptedSteps + 1) = x + h
579 yint(:,me%AcceptedSteps + 1) = y2
580 end if
581
582 ! get ready for the next iteration
583 x = x + h
584 y1 = y2
585
586 ! compute the new step size now
587 if (.NOT. conststepsz) then
588 hnew = stepszhairer(h, .false., me%q, err, stepszparams)
589 ! make sure hnew is not greater than hmax
590 if (abs(hnew) > hmax) hnew = hsign*hmax
591 ! if the last step was rejected, then dont increase step size
592 if (laststeprejected .EQV. .true.) hnew = hsign*min(abs(hnew),abs(h))
593 ! reset
594 laststeprejected = .false.
595 else
596 hnew = h
597 end if
598
599 else
600
601 ! step is rejected, now what
602
603 ! increment the reject step counter, ignore the initial step rejections
604 if (me%AcceptedSteps >= 1) me%RejectedSteps = me%RejectedSteps + 1
605
606 ! compute the new step size now
607 hnew = stepszhairer(h, .true., me%q, err, stepszparams)
608
609 ! Make note of your rejection. It is used in the next iteration
610 ! for computing the new step size.
611 laststeprejected = .true.
612
613 ! Retry again if this was the last step
614 last_step = .false.
615
616 end if
617
618 ! update the step size for next iteration
619 h = hnew
620
621 ! Take another step only if step size is not too small and it is
622 ! not the last step
623 ! Reference: Hairer's DOP853 code but added the last step check
624 if ((.NOT. conststepsz) .AND. (.NOT. last_step) &
625 .AND. abs(h)*0.01_wp <= abs(x)*eps) then
626 status = flint_error_stepsz_toosmall
627 end if
628
629 end do
630
631 ! We have either accomplished our job or finished prematurely?!
632
633 ! return the solution at the integrator's internal steps
634 if (intstepsneeded) then
635 block
636 real(WP), allocatable, dimension(:) :: Xarr
637 real(WP), allocatable, dimension(:,:) :: Yarr
638
639 allocate(xarr(me%AcceptedSteps + 1))
640 allocate(yarr(pdiffeqsys%n,me%AcceptedSteps + 1))
641
642 xarr = xint(1:me%AcceptedSteps + 1)
643 yarr = yint(:,1:me%AcceptedSteps + 1)
644
645 call move_alloc(from=xarr, to=xint)
646 call move_alloc(from=yarr, to=yint)
647 end block
648 end if
649
650
651 ! Update internal storage size based on accepted steps
652 if (interpon) then
653 block
654 real(WP), allocatable, dimension(:) :: Xarr
655 !real(WP), allocatable, dimension(:,:) :: Yarr
656 real(WP), allocatable, dimension(:,:,:) :: Biparr
657
658 allocate(xarr(me%AcceptedSteps + 1))
659 allocate(biparr(ninterpstates,0:me%pstar,me%AcceptedSteps))
660
661 xarr = me%Xint(1:me%AcceptedSteps + 1)
662 biparr = me%Bip(:,:,1:me%AcceptedSteps)
663
664 call move_alloc(from=xarr, to=me%Xint)
665 !call move_alloc(from=Yarr, to=me%Yint)
666 call move_alloc(from=biparr, to=me%Bip)
667 end block
668 end if
669
670 ! Prepare event output data
671 if (eventson .AND. allocated(eventdata)) then
672 eventstates = reshape(eventdata, [pdiffeqsys%n+2, int(size(eventdata)/(pdiffeqsys%n+2))])
673 end if
674
675 if (last_step .AND. status == flint_success) then
676 ! we finished like we should
677 elseif (eventson .AND. last_step .AND. status == flint_event_term) then
678 ! One of the terminal event has triggered, do nothing
679 else if (me%TotalSteps >= me%MaxSteps .AND. status == flint_success) then
680 ! maximum steps reached
681 status = flint_error_maxsteps
