28 module subroutine erk_interp(me, xarr, yarr, saveinterpcoeffs)
29
30 implicit none
31
32 class(ERK_class), intent(inout) :: me
33
34 real(WP), dimension(1:), intent(in) :: Xarr
35 real(WP), dimension(size(me%InterpStates),size(Xarr)), intent(out) :: Yarr
36 logical, intent(in), optional :: SaveInterpCoeffs
37
38 logical :: DeallocMem
39 integer :: n, X0loc, X0start
40 real(WP) :: h, X0
41 real(WP) :: hSign
42 integer :: ctr
43
44 ! if interpolation mode is not on then return with an error
45 if (.NOT. me%InterpOn) then
46 me%status = flint_error_intp_off
47 return
48 end if
49
50 ! if user does not explicitly say to keep the internal memory for future interpolations
51 ! then free this memory.
52 deallocmem = .true.
53 if (present(saveinterpcoeffs)) deallocmem = (.NOT. saveinterpcoeffs)
54
55 ! If empty array then just return without error
56 n = size(xarr)
57 if (n < 1) return
58
59 ! Xarr must be monotonically increasing or decreasing with no repetitions
60 hsign = sign(1.0_wp, (me%Xint(me%AcceptedSteps+1)-me%Xint(1)))
61 block
62 real(WP), dimension(size(Xarr)-1) :: tXarr
63 txarr = xarr(1:(n-1)) - xarr(2:n)
64 if (any((hsign*txarr) >= 0.0_wp)) then
65 me%status = flint_error_intp_array
66 return
67 end if
68 end block
69
70 ! All Xarr elements must belong to [x0, xf] closed set but we need to
71 ! check the first and last elements as this is a sorted array.
72 if (hsign*(xarr(1)-me%Xint(1)) < 0.0_wp &
73 .OR. hsign*(xarr(n)-me%Xint(me%AcceptedSteps+1)) > 0.0_wp) then
74 me%status = flint_error_intp_array
75 return
76 end if
77
78 ! Binary search seems to be slower than the linear search. May be
79 ! because Xarr is also a sorted array
80 ! Xarrloc = BinarySearch(Xarr, me%Xint)
81
82 ! Main Interpolation loop
83 x0start = 1
84
85 do concurrent(ctr = 1:n)
86
87 ! Find X0loc s.t. Xarr(ctr) belongs to ( Xint(X0loc-1), Xint(X0loc) ]
88 do while (x0start <= me%AcceptedSteps)
89 x0start = x0start + 1
90 if (hsign*(me%Xint(x0start)-xarr(ctr)) >= 0.0_wp) then
91 x0loc = x0start - 1
92 exit
93 end if
94 end do
95
96 if (x0loc <= me%AcceptedSteps) then
97 x0 = me%Xint(x0loc)
98 h = me%Xint(x0loc+1)-me%Xint(x0loc)
99 else
100 ! There is a problem, we should never reach here!
101 x0 = xarr(ctr)
102 h = 0.0_wp
103 end if
104
105 ! compute the interpolation polynomial
106 yarr(:, ctr) = interpy(size(me%InterpStates), xarr(ctr), x0, h, &
107 me%pstar, me%Bip(:,:,x0loc))
108
109 ! prepare for the next point
110 x0start = x0loc
111 end do
112
113 ! job done, deallocate memory
114 if (deallocmem .EQV. .true.) then
115 deallocate(me%Xint)
116 !deallocate(me%Yint)
117 deallocate(me%Bip)
118 end if
119
120
121 end subroutine erk_interp

124 pure module function interpy(np, x, x0, h, pstar, bcoeffs) result(yip)
125 integer, intent(in) :: np
126 real(WP), intent(in) :: X, X0, h
127 integer, intent(in) :: pstar
128 real(WP), dimension(np,0:pstar), intent(in) :: bCoeffs
129 real(WP), dimension(np) :: Yip
130
131 real(WP) :: theta
132 integer :: k
133
134 ! compute the interpolation polynomial
135 theta = (x - x0)/h
136 yip = 0.0_wp
137 do k = 0,pstar
138 yip = yip + bcoeffs(:,k)*(theta**k)
139 end do
140 end function interpy
